#include <iostream>
#include "generateAddition.h" 
#include <time.h>       /* time */
using namespace std;

generateAddition::generateAddition()
{
	/* initialize random seed: */
  srand (time(NULL));

  /* generate secret numbers between 1 and 10: */
  num1 = rand() % 10 + 1;
  num2 = rand() % 10 + 1;
  theAnswer = num1 + num2;
}

generateAddition::generateAddition(int& number1, int& number2)
{
	/* initialize random seed: */
  srand (time(NULL));

  /* generate secret numbers between 1 and 10: */
  num1 = rand() % 10 + 1;
  num2 = rand() % 10 + 1;
  theAnswer = num1 + num2;
  number1 = num1;
  number2 = num2;
}
